#include

void main()
{
	FILE* fpr;//文件读取的指针
	char pathr[] = "C:\\Users\\Administrator\\Desktop\\test1.txt";
	fpr = fopen(pathr,"r");//按照路径读取
	if (fpr==NULL)
	{
		printf("打开读取的文件失败\n");
	}
	else
	{
		while (!feof(fpr))//feof(fpr)如果文件结束，则返回非0值，否则返回0
		{
			char ch = fgetc(fpr);//读取字符
			putchar(ch);//输出字符
		}
	}
	fclose(fpr);//关闭输入文件的流
}

2、文件复制

将test1的文件内容复制到test2


#define _CRT_SECURE_NO_WARNINGS//去掉安全检查
#include
#include
//复制文本文件
void main()
{
	FILE* fpr;//文件读取的指针
	FILE* fpw;//文件写入的指针
	char pathr[] = "C:\\Users\\Administrator\\Desktop\\test1.txt";
	char pathw[] = "C:\\Users\\Administrator\\Desktop\\test2.txt";
	fpr = fopen(pathr,"r");//按照路径读取
	fpw = fopen(pathw, "w");//按照路径写入
	if (fpr==NULL)
	{
		printf("打开读取的文件失败\n");
		return;
	}
	if (fpw == NULL)
	{
		printf("打开写入的文件失败\n");
		return;
	}
	while (!feof(fpr))//feof(fpr)如果文件结束，则返回非0值，否则返回0
	{
		char ch = fgetc(fpr);//读取字符
		putchar(ch);//输出字符
		fputc(ch,fpw);//写入字符
	}
	fclose(fpr);//关闭输入文件的流
	fclose(fpw);//关闭输出文件的流
}

3、实现将录入的通讯录信息保存到磁盘文件


#include
#include
struct address_list                        //定义结构体存储学生信息
{
	char cName[10];
	char cAdr[20];
	char cTel[15];
}info[100];
void save(char *name, int n)               //自定义save函数
{
	FILE *fp;                              //定义一个指向FILE类型结构体的指针变量
	int i;
	if ((fp=fopen(name,"wb"))==NULL)       //以只写方式打开指定文件
	{
		printf("cannot open file\n");
		exit(0);
	}
	for ( i = 0; i < n; i++)
	{
		if (fwrite(&info[i],sizeof(struct address_list),1,fp)!=1)//将一组数据输出到fp所指定的文件中
		{
			printf("file write erroe\n");//如果写入文件不成功，则输出错误
		}
	}
	fclose(fp);//关闭文件
}
void show(char *name, int n)
{
	int i;
	FILE *fp;
	if ((fp=fopen(name,"rb"))==NULL)//以只读方式打开指定文件
	{
		printf("cannot open file\n");
		exit(0);
	}
	for ( i = 0; i < n; i++)
	{
		fread(&info[i], sizeof(struct address_list), 1, fp);
		printf("%15s%20s%20s\n",info[i].cName,info[i].cAdr,info[i].cTel);
	}
	fclose(fp);
}
main()
{
	int i, n;
	char filename[50];
	printf("how many?\n");
	scanf("%d",&n);
	printf("please input filename:\n");
	scanf("%s",filename);//输入文件所在路径及名称
	printf("please input name,addresas,telephone:\n");
	for ( i = 0; i < n; i++)
	{
		printf("NO%d",i+1);
		scanf("%s%s%s",info[i].cName,info[i].cAdr,info[i].cTel);
		save(filename,n);
	}
	show(filename,n);
}
